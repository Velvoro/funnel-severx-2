📋 Schéma SQL Supabase

Exécutez ce script dans Supabase > SQL Editor :

CREATE TABLE IF NOT EXISTS funnel_config (
  id BIGSERIAL PRIMARY KEY,
  funnel_id TEXT UNIQUE NOT NULL,
  config JSONB NOT NULL DEFAULT '{}',
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS funnel_events (
  id BIGSERIAL PRIMARY KEY,
  funnel_id TEXT,
  visitor_id TEXT,
  event_type TEXT,
  event_data JSONB DEFAULT '{}',
  page_url TEXT,
  user_agent TEXT,
  referrer TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_events_funnel_created
  ON funnel_events(funnel_id, created_at);

ALTER TABLE funnel_events ENABLE ROW LEVEL SECURITY;
ALTER TABLE funnel_config ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Public insert events" ON funnel_events FOR INSERT WITH CHECK (true);
CREATE POLICY "Admin read events" ON funnel_events FOR SELECT USING (true);
CREATE POLICY "Admin all config" ON funnel_config FOR ALL USING (true);
